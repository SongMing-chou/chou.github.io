进程间通信

关于进程间通信自己的测试代码的简单实现。

管道 （用户空间）

无名管道 pip(fd[2]) 父子进程

//by Chou
/*
output
pid == 16310, hello linux!!!
pid == 0, hello linux!!!
*/
#include<iostream>
#include<unistd.h>
#include<stdlib.h>
#include<stdio.h>
/*int pipe(int fd[2]); f[1]xie f[0]du
如果写文件符引用计数为0，一进程读的时候 直接返回 0
*/
int main(int argc,char * argv[]) {
    int fd[2];
    int ret = 0;
    
    char write_buff[] = "hello linux!!!";
    char read_buff[128] ={0};

    ret = pipe(fd);
    if(ret < 0) {
        printf("create pipe fail!!\n");
        return -1;
    }

    int m_pid = fork();
    //子进程
    if(m_pid == 0) {
        // close(fd[1]);测试引用计数为0时的行为，
        // close(fd[0]);
        int n = read(fd[0],&read_buff,sizeof(write_buff));//阻塞
        if(n < 0) {
            printf("read fail !!\n");
        }
        printf( "pid == %d, %s\n",m_pid ,read_buff);
    } else {
        // close(fd[1]);同上
        // close(fd[0]);
        int n = write(fd[1],&write_buff,sizeof(write_buff));//管道满了还继续写会阻塞  大小默认65536
        if(n < 0) {
            printf("write faill!!\n");
        }else{
            printf("pid == %d, %s\n",m_pid,write_buff);
        }
        sleep(5);
    }
    close(fd[0]),close(fd[1]);//关闭引用计数为0的文件描述符 行为是未定义的，万一恰好创建一个值一样的文件描述符？
    return 0;
}

命名管道

创建了一个真实文件 FIFO文件，要传入路径名。

#include<stdio.h>
#include<stdlib.h>
#include<unistd.h>
#include<sys/types.h>
#include<fcntl.h>
#include<sys/stat.h>

//acess(cost char *pathname,int mode) mode == 0用来判断文件是否存在
// #define MAIN1
// #define MAIN2
//使用g++ -o mymkfifo mymkfifi.cpp -DMAIN1 or -DMAIN2分别编译
int main(int argc,char *argv[]) {
    int fd;
    char c;
    //server
    #ifdef MAIN1
    if(access("./choufifo",0) < 0) {
        int n = mkfifo("./choufifo",0777);
        if(n < 0) {
            printf("create fifo fail!!!\n");
            return -1;
        }else{
            printf("create fifo sucess!!\n");
        }
    }
    fd = open("./choufifo",O_WRONLY);//设置阻塞的话，阻塞到有进程读打开，非阻塞直接返回-1，且不会打开
    if(fd < 0) {
        printf("open fifo on write fail!!!\n");
    }

    while(c!='q'&&c!='Q') {
        scanf("%c",&c);
        write(fd,&c,sizeof(c));
    }
    #endif
    //client
    #ifdef MAIN2
    while(access("./choufifo",0) < 0) {
        printf("wait main1 create fifo\n");
        sleep(1);
    }
    fd = open("./choufifo",O_RDONLY);//设置阻塞的话，阻塞到有进程写打开，非阻塞直接返回-1，且不会打开
    if(fd < 0) {
        printf("open fifo on read fail!!!\n");
    }
    while(c!='q'&&c!='Q') {
        read(fd,&c,sizeof(c));
        printf("main2 get char = %c from fifo!!!\n",c);
        printf("\n");
    }
    #endif
    while(c!='q'&&c!='Q');
    return 0;
}

IPC通信 (内核空间)

共享内存，消息队列，信号。
三种通信都需要借助key值 

// key值：因为我们需要一个唯一一个IPC对象的身份ID，(IPC_id 是由操作系统来管理的)，但是这个ID只能被创建他的进程获取；
key_t ftok(const char*pathname,int proj_id);//
/*
1、ftok根据路径名，提取文件信息，再根据这些文件信息及project ID合成key，该路径可以随便设置。
2、该路径是必须存在的，ftok只是根据文件inode在系统内的唯一性来取一个数值，和文件的权限无关。
3、proj_id是可以根据自己的约定，随意设置。这个数字,有的称之为project ID; 在UNIX系统上,它的取值是1到255。
4、多进程时，在ftok是需要保证路径存在不被删除，更改，否则生成的key不一样，调用shmget生成的共享内存就不一样了。
*/

共享内存

所谓共享内存是指多个进程都可以访问的同一块地址空间，但是我们知道Linux操作系统为了保证系统执行的安全，
为每个进程划分了各自独立的地址空间，每个进程不能访问别的进程的地址空间，那么共享内存实现的原理是什么呢？​
内核开辟一块物理内存区域，进程本身将这片内存空间映射到自己的地址空间进行读写。​
进程可以直接访问这片内存，数据不需要在两进程间复制，所以速度较快。共享内存没有任何的同步与互斥机制，
所以要使用信号量来实现对共享内存的存取的同步。​
当需要使用共享内存进行通信时，一般步骤如下：

  • 先创建一片共享内存，该内存存在于内核空间中。
  • 进程通过key值找到这片共享内存的唯一ID，然后将这片共享内存映射到自己的地址空间。
  • 每个进程通过读写映射后的地址，来访问内核中的共享内存。

看代码更直观

//终端1：g++ -o shmemery1 shmemery.cpp -DMAIN1 && ./shmemery1 分别编译
//终端2：g++ -o shmemery2 shmemery.cpp -DMAIN2 && ./shmemery2
#include<stdio.h>
#include<stdlib.h>
#include<unistd.h>
#include<sys/types.h>
#include<sys/shm.h>
#include<signal.h>
#include<string>
/*
步骤
1.key_t ftok(const char*path,char key); 生成key当然用 IPC_PRIVATE也是可以的 但是可能会出错
//内核空间开辟内存用于共享
2.int shmget(key_t key, int size,int shmflg);
    key ftok返回值也可以是IPC_PRIVATE == 0
    size 大小
    shmflg 同open一样的权限位1 2 4

//将内核的共享内存映射到用户空间 失败返回null成功返回地址
3.void*shmat(int shmid,const void*shmaddr,int shmflg); 
    shmid shmget返回值
    shmaddr 映射地址 NULL为系统自动映射
    shmflg 权限位 0为可读可写

//共享内存管理
4.int shmctl(int shmid, int cmd, struct shmid_ds *buf);

//断开共享内存连接
5.int shmdt(const void *shmaddr)；
    shmdt 连接共享内存的起始地址 shmat返回值
*/
//下面是两个cs模型验证
int main(int argc,char *argv[]) {

    int pid,key,shmid;

    key = ftok("./chou_key",10);
    if(key < 0) {
        #ifdef MAIN1
        printf("server create key faill!!\n");
        #endif
        #ifdef MAIN2
        printf("client create key faill!!\n");
        #endif
    }
    shmid = shmget(key, 128, IPC_CREAT | 0777);//大小为128
    if(shmid < 0) {
        #ifdef MAIN1
        printf("server shmget  faill!!\n");
        #endif
        #ifdef MAIN2
        printf("client shmget  faill!!\n");
        #endif
    }
    char *p  = (char*)shmat(shmid, NULL, 0);
    if(p == NULL) {
        #ifdef MAIN1
        printf("server shmat  faill!!\n");
        #endif
        #ifdef MAIN2
        printf("client shmat  faill!!\n");
        #endif
    }
    char c;
    char * temp = p;
#ifdef MAIN1
    for(int i = 0;i < 128;i++) {
        *temp = i+30;
        temp++;
    }
    while(c!='Q'&&c!='q'){
        scanf("%c",&c);
    }
#endif
#ifdef MAIN2
    printf("data from server:\n");
    for(int i = 0;i < 128;i++) {
        printf("%c ",*temp);
        temp++;
    }
    printf("\n");
    while(c!='Q'&&c!='q'){
        scanf("%c",&c);
    }
#endif
    shmdt(p);
    shmctl(shmid,IPC_RMID,NULL);
    return 0;
}

消息队列

和管道差不多但是：
1.生命周期跟随内核，消息队列一直存在，需要用户显示调用接口删除或者使用命令删除。
2.消息队列可以实现双向通信。
3.克服了管道只能承载无格式字节流的缺点。

#include<stdio.h>
#include<stdlib.h>
#include<unistd.h>
#include<sys/types.h>
#include<sys/msg.h>
#include<signal.h>
#include<string.h>


/*c++
1.
创建或者打开一个已有消息队列的函数
头文件：#include <sys/types.h>
        #include <sys/ipc.h>
       #include <sys/msg.h>
int msgget(key_t key, int flag)
     key 和消息队列关联的key值 和msgget一样 标识一个全局唯一的消息队列。
     flag 消息队列的访问权限
返回值： 成功，消息队列ID，出错 -1并设置errno
如果msgget用于创建消息队列，则与之关联的内核数据结构msqid_ds将被创建并初始化，msqid_ds的结构体定义如下：
struct msqid_ds {
    struct ipc_perm_msg_perm;   //消息队列的操作权限
    time_t msg_stime;           //最后一次调用msgsend的时间
    time_t msg_rtume;           //最后一次调用msgrcv的时间
    time_t msg_ctime;           //最后一次被修改的时间
    unsigned long __msg_cbytes; //消息队列中已有的字节数
    msgqnum_t msg_qnum;         //消息队列中已有的消息数
    msglen_t msg_qbytes;        //消息中允许最大的消息数
    pid_t msg_lsg_lspid;        //消息最后执行msgsnd的进程的PID
    pid_t msg_lrpid;            //最后执行msgrcv的进程的PID
}
//系统把一条消息添加到消息队列中，其定义如下
2.int msgsnd (int msqid,const void*msq_ptr,size_t msg_sz,int msgflg);
参数 
    msqid：是由msgget调用返回的消息队列标识符
    msq_ptr:发送的消息
    struct msgbuf {
        long mtype; //消息类型
        char mtext[512]；//消息数据
    }
    msg_sz:是消息数据部分(mtext)的长度，长度可以为0，表示消息没有数据
    msgflg:控制msgsnd的行为，通常只支持IPC_NOWAIT标志，即非阻塞；如果消息队列满了，msgsnd将阻塞，若指定非阻塞
    则msgsnd将返回并设置errno为EAGAIN
    该函数处于阻塞的调用也是可以被中断的：
    1. 消息队列被移除，立即返回设置errno 为EIDRM
    2. 程序接收到信号，立即返回并设置errno为EINTR
    成功时会修改msqid_ds的部分字段
    1.msg_qnum 加1
    2.msg_lspid设置为当前调用的pid
    3.msg_stime设置为当前的时间
3.int msgrcv(int msqid,void*msg_ptr,size_t msg_sz,long int msgtype,int msgflg);
参数：
    msqid 由msgget调用返回的消息队列标识符，
    msg_ptr参数用于存储接收的消息，
    msg_sz参数是指消息数据部分的长度，
    msgtype参数指定接收何种类型的消息，如下：
    1.msgtype == 0 读取消息队列中的第一个消息
    2.msgtype > 0 读取消息队列中第一类型为msgtype的消息
    3.msgtype < 0 读取消息队列中第一个类型绝对值比msgtype小的消息
    msgtype控制msgrcv函数的行为，它可以是以下标志的按位或
    1 IPC_NOWAIT 如果消息队列中没有消息，则msgrcv调用立即返回并设置errno为ENOMSG。
    2.MSG_EXCEPT，如果msgtype大于0，则接收消息队列中第一个非msgtype类型的消息
    3.MSG_NOERROR，如果消息数据部分长度超过了msg_sz，就将他截断。
        该函数处于阻塞的调用也是可以被中断的：
    1. 消息队列被移除，立即返回设置errno 为EIDRM
    2. 程序接收到信号，立即返回并设置errno为EINTR
    成功时会修改msqid_ds的部分字段
    1.msg_qnum 减 1
    2.msg_lspid设置为当前调用的pid
    3.msg_stime设置为当前的时间
//控制消息队列某些属性
4.int msgctl(int msqid,int command,struct msqid_ds*buf);
*/

//下面是两个cs模型验证

    int mtype = 100;
    char mtext[128] = "ppppppp";

};
int main(int argc,char *argv[]) {

    int pid,key,msgid;

    key = ftok("./chou_key",10);
    if(key < 0) {
        #ifdef MAIN1
        printf("server create key faill!!\n");
        #endif
        #ifdef MAIN2
        printf("client create key faill!!\n");
        #endif
        return -1;
    }
    msgid = msgget(key, IPC_CREAT| 0777);
    if(msgid < 0) {
        #ifdef MAIN1
        printf("server msgget  faill!!\n");
        #endif
        #ifdef MAIN2
        printf("client msgget  faill!!\n");
        #endif
        return -1;
    }
    printf("msgid == %d\n",msgid);
#ifdef MAIN1
    struct msgbuf1 send_buf;
    while(1) {
        memset(send_buf.mtext,0,128);
        printf("put massege  ?\n");
        fgets(send_buf.mtext,128,stdin);
        printf("the mesg level is ?\n");
        int c ;
        scanf("%d",&c);
        send_buf.mtype = c;
        printf("set send_buf level  = %d mtext  = %s\n",send_buf.mtype,send_buf.mtext);
        msgsnd(msgid,(void*)&send_buf,strlen(send_buf.mtext),0);
    }
    // while(1);
#endif
#ifdef MAIN2
    struct msgbuf1 rev_buf;
    system("ipcs -q");
    while(1) {
        memset(rev_buf.mtext,0,128);
        printf("want the mesg level is?\n");
        int c;
        scanf("%d",&c);
        rev_buf.mtype = c;
        printf("set rev_buf level  = %d\n",rev_buf.mtype);
        msgrcv(msgid, (void *)&rev_buf, 128,rev_buf.mtype,IPC_NOWAIT);
        printf("data from server:\n");
        printf("%s\n",rev_buf.mtext);
    }
    // while(1);
#endif
    return 0;
}
